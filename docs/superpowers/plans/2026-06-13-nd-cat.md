```python
"""Generic N-D concat (tt.cat rank>=2) via the closed-form direct-copy template.
Mirrors upstream test_cat_nd. Serial GPU."""
import pytest

try:
    import torch
    import triton
    import triton.language as tl
    from triton.tools.tensor_descriptor import TensorDescriptor
    import Metal
    HAS = Metal.MTLCreateSystemDefaultDevice() is not None
except Exception:
    HAS = False

requires_metal = pytest.mark.skipif(not HAS, reason="Metal/torch/triton needed")

if HAS:
    @triton.jit
    def _cat_nd(x_desc, y_desc, z_desc, dim: tl.constexpr, shape: tl.constexpr):
        rank: tl.constexpr = len(shape)
        x = x_desc.load([0] * rank)
        y = y_desc.load([0] * rank)
        z = tl.cat(x, y, dim=dim)
        z_desc.store([0] * rank, z)

    def _alloc(size, align, stream):
        return torch.empty(size, dtype=torch.int8, device="cpu")


@requires_metal
@pytest.mark.parametrize("shape,dim", [
    ((8, 16), 0), ((8, 16), 1),          # 2-D, both axes
    ((4, 4, 16), 2),                     # 3-D last axis
    ((4, 8, 32), 1),                     # 3-D middle axis, out=4*16*32=2048>1024
])
def test_nd_cat(shape, dim):
    triton.set_allocator(_alloc)
    x = torch.rand(shape); y = torch.rand(shape)
    z_ref = torch.cat([x, y], dim=dim)
    z = torch.empty_like(z_ref)
    xd = TensorDescriptor.from_tensor(x, block_shape=list(shape))
    yd = TensorDescriptor.from_tensor(y, block_shape=list(shape))
    zd = TensorDescriptor.from_tensor(z, block_shape=list(z_ref.shape))
    _cat_nd[(1,)](xd, yd, zd, dim=dim, shape=shape)
    torch.testing.assert_close(z, z_ref, atol=0, rtol=0)
```

```python
    def _detect_nd_cat(self):
        """Detect a generic N-D concat: one tt.cat of rank>=2, its two operands
        each tracing (via reshape/convert_layout) to a tt.load, the result
        feeding a tt.store, no reduce/scan/dot/control-flow. Emits a closed-form
        direct copy. Returns dict or None."""
        cat_ssa = store_ssa = None
        loads = []
        for ssa in self.graph.ops:
            if ssa.op == "tt.load":
                loads.append(ssa)
            elif ssa.op == "tt.store":
                if store_ssa is not None:
                    return None
                store_ssa = ssa
            elif ssa.op == "tt.cat":
                if cat_ssa is not None:
                    return None
                cat_ssa = ssa
            elif ssa.op in ("tt.reshape", "ttg.convert_layout", "tt.join"):
                if ssa.op == "tt.join":
                    return None  # join handled elsewhere / refused
            elif ssa.op in ("scf.for", "scf.while", "scf.if",
                            "tt.reduce", "tt.scan", "tt.dot"):
                return None
        if cat_ssa is None or store_ssa is None or len(loads) < 2:
            return None
        if len(cat_ssa.operand_ids) < 2:
            return None
        src_shape = _extract_shape(self._find_op_type_str(cat_ssa.operand_ids[0]))
        dst_shape = _extract_shape(self._find_op_type_str(cat_ssa.id)
                                   or cat_ssa.type_str)
        if (not src_shape or not dst_shape or len(src_shape) < 2
                or len(src_shape) != len(dst_shape)):
            return None
        # Infer dim: the single axis where dst == 2*src (others equal).
        dims = [d for d in range(len(src_shape))
                if dst_shape[d] == 2 * src_shape[d]]
        others_ok = all(dst_shape[d] == src_shape[d]
                        for d in range(len(src_shape)) if d not in dims)
        if len(dims) != 1 or not others_ok:
            return None
        dim = dims[0]

        # Data-flow guard: each cat operand traces (reshape/convert only) to a
        # load; the cat result reaches the store the same way.
        load_ids = {l.id for l in loads}
        _VP = ("tt.reshape", "ttg.convert_layout")

        def _traces_to_set(sid, targets):
            cur, seen = sid, set()
            for _ in range(8):
                if cur in targets:
                    return True
                if cur in seen:
                    break
                seen.add(cur)
                op = next((s for s in self.graph.ops if s.id == cur), None)
                if op is None or op.op not in _VP or not op.operand_ids:
                    break
                cur = op.operand_ids[0]
            return False

        if not all(_traces_to_set(oid, load_ids)
                   for oid in cat_ssa.operand_ids[:2]):
            return None
        if (len(store_ssa.operand_ids) < 2
                or not _traces_to_set(store_ssa.operand_ids[1], {cat_ssa.id})):
            return None

        ptr_args = [a for a in self.graph.args if a.is_ptr]
        if len(ptr_args) < 3:
            return None
        total = 1
        for s in dst_shape:
            total *= s
        return {
            "x_arg": ptr_args[0].name,
            "y_arg": ptr_args[1].name,
            "out_arg": ptr_args[2].name,
            "elem_type": ptr_args[2].elem_type,
            "src_shape": list(src_shape),
            "dst_shape": list(dst_shape),
            "dim": dim,
            "total": total,
        }
```

```python
    def _lower_nd_cat_template(self, info) -> str:
        """Closed-form N-D concat: out[k] = (O[dim]<D) ? x[flat] : y[flat]."""
        src_shape = info["src_shape"]; dst_shape = info["dst_shape"]
        dim = info["dim"]; total = info["total"]
        D = src_shape[dim]
        rank = len(src_shape)

        def _strides(shape):
            st = [1] * len(shape)
            for i in range(len(shape) - 2, -1, -1):
                st[i] = st[i + 1] * shape[i + 1]
            return st

        dst_stride = _strides(dst_shape)
        src_stride = _strides(src_shape)
        od = (f"(k % {dst_shape[dim]}u)" if dst_stride[dim] == 1
              else f"((k / {dst_stride[dim]}u) % {dst_shape[dim]}u)")
        # input flat: O[d]*src_stride[d], with O[dim] = od % D (folds the y-half).
        terms = []
        for d in range(rank):
            if d == dim:
                terms.append(f"((({od}) % {D}u)) * {src_stride[d]}u")
            else:
                o_d = (f"(k % {dst_shape[d]}u)" if dst_stride[d] == 1
                       else f"((k / {dst_stride[d]}u) % {dst_shape[d]}u)")
                terms.append(f"{o_d} * {src_stride[d]}u")
        in_flat = " + ".join(terms)

        x_arg = info["x_arg"]; y_arg = info["y_arg"]; out_arg = info["out_arg"]
        safe_name = _sanitize_msl_name(self.graph.func_name)
        num_warps = self.options.num_warps if self.options else 4
        threads = num_warps * 32
        arg_decls = []
        for i, arg in enumerate(self.graph.args):
            if arg.is_ptr:
                arg_decls.append(
                    f"    device {triton_type_to_msl(arg.elem_type)}* "
                    f"{arg.name} [[buffer({i})]]")
            else:
                amt = (triton_type_to_msl(arg.elem_type) if arg.elem_type
                       else "int")
                arg_decls.append(
                    f"    constant {amt}& {arg.name} [[buffer({i})]]")
        lines = [
            "#include <metal_stdlib>", "using namespace metal;", "",
            f"kernel void {safe_name}(",
            ",\n".join(arg_decls) + ",",
            "    uint pid [[threadgroup_position_in_grid]],",
            "    uint lid [[thread_position_in_threadgroup]],",
            "    uint tid [[thread_position_in_grid]]",
            ") {",
            f"    for (uint k = lid; k < {total}u; k += {threads}u) {{",
            f"        {out_arg}[k] = (({od}) < {D}u) ? "
            f"{x_arg}[{in_flat}] : {y_arg}[{in_flat}];",
            "    }",
            "}", "",
        ]
        return "\n".join(lines)
```

```python
def _check_nd_cat_join(ctx: RefusalContext) -> Optional[Violation]:
    _VP = ("tt.reshape", "ttg.convert_layout")
    by_id = {s.id: s for s in ctx.all_ops}
    load_ids = {s.id for s in ctx.all_ops if s.op == "tt.load"}
    store_val_ids = {s.operand_ids[1] for s in ctx.all_ops
                     if s.op == "tt.store" and len(s.operand_ids) >= 2}

    def _traces(sid, targets):
        cur, seen = sid, set()
        for _ in range(8):
            if cur in targets:
                return True
            if cur in seen:
                break
            seen.add(cur)
            o = by_id.get(cur)
            if o is None or o.op not in _VP or not o.operand_ids:
                break
            cur = o.operand_ids[0]
        return False

    for s in ctx.all_ops:
        if s.op in ("tt.cat", "tt.join") and s.operand_ids:
            sh = ctx.extract_shape(
                ctx.find_op_type_str(s.operand_ids[0]) or "")
            if sh and len(sh) >= 2:
                # Defer the clean N-D tt.cat copy pattern to _detect_nd_cat:
                # operands trace to loads AND the result reaches a store, via
                # value-preserving ops only. join + non-clean cat still refuse.
                if (s.op == "tt.cat"
                        and all(_traces(oid, load_ids)
                                for oid in s.operand_ids[:2])
                        and any(_traces(v, {s.id}) for v in store_val_ids)):
                    continue
                return Violation(
                    f"'{s.op}' on a rank-{len(sh)} tensor is not "
                    "supported (the generic handler is 1-D only); an "
                    "N-D concat/join would be laid out incorrectly "
                    "(test_cat_nd).", s.op)
    return None
```
